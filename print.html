(consp '())
;;=> NIL

;;=> T

(consp '(1 . 2))
;;=> T

(consp '(1 2))
;;=> T

(type-of '())
;;=> NULL
(typep '() 'atom)
;;=> T

(type-of '(1))
;;=> CONS


(typep '() 'cons)
;;=> NIL
(typep '() 'list)
;;=> T

(typep '(1) 'cons)
;;=> T
(typep '(1) 'list)
;;=> T

(typep '(1 . 2) 'cons)
;;=> T
(typep '(1 . 2) 'list)
;;=> T

배열

(defparameter *array* (make-array '(2 4) :initial-element 0))
;;=> *ARRAY*

*array*
;; => #2A((0 0 0 0) (0 0 0 0))

(aref *array* 0 0)
;; => 0

(setf (aref *array* 0 0) 100)
;; => 100

*array*
;; => #2A((100 0 0 0) (0 0 0 0))

(setf (aref *array* 1 1) 100)
;; => 100

*array*
;; => #2A((100 0 0 0) (0 100 0 0))
